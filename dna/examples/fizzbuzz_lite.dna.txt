; FizzBuzz-lite: for 1-15, output special values
; 0 = fizz (div by 3), -1 = buzz (div by 5), -2 = fizzbuzz (both)
; else output the number itself
fn check(n) do
  if n % 15 == 0 do
    return -2
  end
  if n % 3 == 0 do
    return 0
  end
  if n % 5 == 0 do
    return -1
  end
  return n
end

let i = 1
while i <= 15 do
  check(i)
  i = i + 1
end
